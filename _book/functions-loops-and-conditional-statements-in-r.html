4 Functions, loops and conditional statements in R

4.1 Writing functions in R

number <- 10
number_increased <- number ^2
print(paste0("The result is: ", number_increased))
## [1] "The result is: 100"
myFun <- function(){
  number <- 10
  number_increased <- number ^2
  return(print(paste0("The result is: ", number_increased)))
}
myFun()
## [1] "The result is: 100"
myFunVar <- function(x){
  number <- x
  number_increased <- number ^2
  return(print(paste0("The result is: ", number_increased)))
}
myFunVar(5)
## [1] "The result is: 25"
myFunVarMulti <- function(x,y){
  number <- x
  number_increased <- number ^y
  return(print(paste0("The result is: ", number_increased)))
}
myFunVarMulti(5, 3)
## [1] "The result is: 125"

Character-function

greetName <- function(name){
  return(print(paste0("Hello ", name)))
}
greetName("Arie")
## [1] "Hello Arie"

4.3 Conditional statements in R

a = 10
b = 20
if (a > b){
  print("A is larger than B")
} else {
  print("That is not true")
}
## [1] "That is not true"
if (a > b){
  print("A is larger than B")
} else if (a < b) {
  print("A is smaller than B")
} else {
  print("That is not true")
}
## [1] "A is smaller than B"
